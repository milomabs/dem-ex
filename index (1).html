        public function login(Request $request){
        $success = auth()->attempt([
        "email" => $request->email,
        "password" => $request->password
        ], request()->has("remember"));
        
        If($success) {
        return redirect()->route("application.index"); }
        }
        
        public function logout(){
        auth()->logout();
        return redirect()->route("auth.login"); 
        }
